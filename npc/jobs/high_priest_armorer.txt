-	script	High Priest Armorer#hpar1::High_Priest_Armorer	-1,{
	set .n$,"[High Priest Armorer]";
	mes .n$;
	mes "Hello! I am an expert in customising armors.";
	mes "I am part of the Priest Guild, so I can make changes to a member's outfit for free";
	next;
	if (Class != Job_High_Priest) {
		mes "[High Priest Armorer]";
		mes "Oh! I can only provide my services to High Priests...";
		mes "Sorry!";
		close;
	}
	mes "Aha!";
	mes "A seasoned High Priest, what outfit would you like to wear?";
	next;
	switch(select("High Priest:Arch Bishop:Pontiff:Cardinal")) {
		case 1:
			setlook LOOK_CLOTHES_COLOR,0;
        	setlook LOOK_BODY2,0;
			specialeffect2 EF_ENTRY2;
			mes "[Lord Knight Armorer]";
			mes "There you go!";
			mes "Come back if you want to change your outfit again!";
			close;
		case 2:
			setlook LOOK_CLOTHES_COLOR,0;
        	setlook LOOK_BODY2,1;
			specialeffect2 EF_ENTRY2;
			mes "[Lord Knight Armorer]";
			mes "There you go!";
			mes "Come back if you want to change your outfit again!";
			close;
		case 3:
			setlook LOOK_CLOTHES_COLOR,0;
        	setlook LOOK_BODY2,2;
			specialeffect2 EF_ENTRY2;
			mes "[Lord Knight Armorer]";
			mes "There you go!";
			mes "Come back if you want to change your outfit again!";
			close;
		case 4:
			setlook LOOK_CLOTHES_COLOR,0;
        	setlook LOOK_BODY2,3;
			specialeffect2 EF_ENTRY2;
			mes "[Lord Knight Armorer]";
			mes "There you go!";
			mes "Come back if you want to change your outfit again!";
			close;;
	}	
}
